 * @ngdoc directive
 * @name latexmadeeasyApp.directive:chapters
 * @description
 * # chapters
 */
 
var ChaptersDirective = function ChaptersDirective() {
  return {
    templateUrl: 'views/chapters.html',
    restrict: 'E',
    controller: 'ChaptersCtrl',
    controllerAs: 'ctrl',
    bindToController: true,
    scope: {
      chapter: '='
    },

    }
  };
};
